async function captureAndProcess() { const video = document.getElementById('vision-feed'); const canvas = document.getElementById('ocr-canvas'); const ctx = canvas.getContext('2d'); // 1. Capture the EXACT moment you hit the button const atomicCaptureTime = new Date(); // 2. G-SHOCK OPTIMIZATION: High contrast & Grayscale // This makes the grey LCD background white and the black segments jet black canvas.width = video.videoWidth; canvas.height = video.videoHeight; ctx.filter = "contrast(160%) grayscale(100%) brightness(110%)"; ctx.drawImage(video, 0, 0); document.getElementById('capture-btn').innerText = "READING..."; // 3. Digital Digit Whitelist // We tell the AI: "Ignore everything except 0-9 and the colon" const worker = await Tesseract.createWorker(); await worker.loadLanguage('eng'); await worker.initialize('eng'); await worker.setParameters({ tessedit_char_whitelist: '0123456789:', }); const { data: { text } } = await worker.recognize(canvas); await worker.terminate(); // Clean up the text (remove spaces/newlines) const cleanText = text.replace(/\s/g, ''); const timeMatch = cleanText.match(/\d{1,2}:\d{2}:\d{2}/); if (timeMatch) { // SUCCESS: Calculate the difference between watch and atomic time const watchSeconds = parseTimeToSeconds(timeMatch[0]); const atomicSeconds = (atomicCaptureTime.getHours() * 3600) + (atomicCaptureTime.getMinutes() * 60) + atomicCaptureTime.getSeconds(); const diff = watchSeconds - atomicSeconds; alert(`SUCCESS!\nWatch: ${timeMatch[0]}\nAtomic: ${atomicCaptureTime.toLocaleTimeString()}\nOffset: ${diff}s`); closeVisionScanner(); } else { alert("STILL BLURRY: Try tilting the watch to avoid glare, or hold it 6 inches away from the lens."); document.getElementById('capture-btn').innerText = "CAPTURE TIME"; } } // Helper to turn HH:MM:SS into total seconds function parseTimeToSeconds(t) { const [h, m, s] = t.split(':').map(Number); return (h * 3600) + (m * 60) + s; }
